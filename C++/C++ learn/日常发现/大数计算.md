## 大数计算

```c++
#include <iostream>
#include <string>
#include <algorithm>
 
using namespace std;
 
string add(string num1, string num2) {
    if (num1.length() < num2.length())
        swap(num1, num2);
 
    int diff = num1.length() - num2.length();
    num2 = string(diff, '0') + num2;
 
    string result = "";
    int carry = 0;
    for (int i = num1.length() - 1; i >= 0; i--) {
        int digit1 = num1[i] - '0';
        int digit2 = num2[i] - '0';
        int sum = digit1 + digit2 + carry;
        carry = sum / 10;
        result += to_string(sum % 10);
    }
 
    if (carry)
        result += to_string(carry);
 
    reverse(result.begin(), result.end());
 
    return result;
}
 
string subtract(string num1, string num2) {
    if (num1.length() < num2.length())
        swap(num1, num2);
 
    int diff = num1.length() - num2.length();
    num2 = string(diff, '0') + num2;
 
    string result = "";
    int borrow = 0;
    for (int i = num1.length() - 1; i >= 0; i--) {
        int digit1 = num1[i] - '0';
        int digit2 = num2[i] - '0' - borrow;
        if (digit1 < digit2) {
            digit1 += 10;
            borrow = 1;
        }
        else {
            borrow = 0;
        }
        int diff = digit1 - digit2;
        result += to_string(diff);
    }
 
    reverse(result.begin(), result.end());
 
    return result;
}
 
string multiply(string num1, string num2) {
    int n1 = num1.size();
    int n2 = num2.size();
    string result(n1 + n2, '0');
 
    for (int i = n1 - 1; i >= 0; i--) {
        int carry = 0;
        for (int j = n2 - 1; j >= 0; j--) {
            int temp = (num1[i] - '0') * (num2[j] - '0') + (result[i + j + 1] - '0') + carry;
            result[i + j + 1] = temp % 10 + '0';
            carry = temp / 10;
        }
        result[i] += carry;
    }
 
    size_t startpos = result.find_first_not_of("0");
    if (string::npos != startpos) {
        return result.substr(startpos);
    }
    return "0";
}
 
string divide(string num1, string num2) {
    string quotient = "0";
    string remainder = num1;
 
    while (remainder.length() >= num2.length()) {
        int shift = remainder.length() - num2.length();
        string temp = num2;
        temp.append(shift, '0');
 
        while (remainder >= temp) {
            remainder = subtract(remainder, temp);
            quotient = add(quotient, "1");
        }
    }
 
    return quotient;
}
 
int main() {
    string num1 = "12345678901234567890";
    string num2 = "98765432109876543210";
 
    cout << "Num1: " << num1 << endl;
    cout << "Num2: " << num2 << endl;
 
    string sum = add(num1, num2);
    cout << "Sum: " << sum << endl;
 
    string difference = subtract(num1, num2);
    cout << "Difference: " << difference << endl;
 
    string product = multiply(num1, num2);
    cout << "Product: " << product << endl;
 
    string quotient = divide(num1, num2);
    cout << "Quotient: " << quotient << endl;
 
    return 0;
}
```
