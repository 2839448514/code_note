> 求三位数上每位数的三次幂之和等于这个数的数有多少个。

```c++
#include <iostream>
#include<cmath>
using namespace std;
vector<int> v;
int main() {
    for(int i=100;i<=1000;i++){
        int a=i%10;
        int c=i/100;
        int b=i/10%10;
        if(pow(a,3)+pow(b,3)+pow(c,3)==i){
            v.push_back(i);
        }
    }
    int n;
}
```
