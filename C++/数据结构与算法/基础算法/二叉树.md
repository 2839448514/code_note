## 二叉树

#### 创建一个二叉树

```c++
#include <iostream>
#include <algorithm>
using namespace std;
struct node {
    int data;
    node *l, *r;
};
void trees(node*& head) {
    int n;
    cin >> n;
    if (n == 0) {
        head = nullptr;
        return;
    }
    head = new node{n, nullptr, nullptr};
    cout<<"输入左子树"<<n<<"节点值："<<endl;
    trees(head->l);
    cout<<"输入右子树"<<n<<"节点值："<<endl;
    trees(head->r);
}
void print_tree(node* head) {
    if (head == nullptr) {
        return;
    }
    cout << head->data << " ";
    print_tree(head->l);
    print_tree(head->r);
}
int main() {
    node *head = nullptr;
    trees(head);
    print_tree(head);
    return 0;
}
```

## 先序遍历

```c++
void print_tree(node* head) {
    if (head == nullptr) {
        return;

    cout << head->data << " ";
    print_tree(head->l);
```

## 中序遍历
