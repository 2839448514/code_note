## 进制计算

1. 二进制中1的个数

```c++
int main(int argc, char* argv[])
{
	int N;
	cin >> N;
	cout << bitset<32>(N) << endl;
	int count = 0;
	while (1) {
		int n = N - 1;
		if (N == 0) {
			cout << "count=" << count << endl;;
			break;
		}
		N = N & n;
		count += 1;
	}
	return 0;
}
```

4. 输出浮点数的二进制

```c++
double double_to_binary(double n) {
	string s;
	s.append("0.");
	while (n!=0) {
		if ((n *= 2) >= 1) {
			s.append("1");
			n = n - 1;
		}
		else {
			s.append("0");
		}
	}
	if (s.length() > 32) {
		cout << "ERROR" << endl;
	}
	else {
		cout << "n=" << s << endl;
	}
	return 0;
}
```
