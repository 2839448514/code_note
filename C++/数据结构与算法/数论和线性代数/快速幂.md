## 快速幂

```c++
#include <iostream>
using namespace std;
typedef long long ll;
ll fastpow(int a,int n,int mod) {
    ll ans = 1;
    a %= mod;
    while (n) {
        if (n & 1) {
            ans = (ans * a) % mod;
        }
            a = (a * a) % mod;
            n >>= 1;
    }
    return ans;
}
int main(int argc, char* argv[]) {
    cout << fastpow(2, 2023, 1000);
    return 0;
}
```
